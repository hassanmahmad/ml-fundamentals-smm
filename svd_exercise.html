import time
import numpy as np
import matplotlib.pyplot as plt
import skimage

# Loading the "cameraman" image
X = skimage.data.camera()
m, n = X.shape

# Compute the SVD decomposition of A and check the shapes
U, s, VT = np.linalg.svd(X, full_matrices=True)
# print(U.shape, s.shape, VT.shape)

# Define the full matrix S
S = np.zeros((m, n))
S[:n, :n] = np.diag(s)

k_values = np.arange(10,100,20)
# print(len(k_values))
compression_values = np.zeros((len(k_values)))

i = 0
for k in k_values:
    Uk = U[:,:k] # m x k
    Sk = S[:k, :k] # k x k
    VTk = VT[:k,:] # k x n

    # compute Xk
    Xk = Uk @ Sk @ VTk
    # time.sleep(1)
    plt.subplot(1, 2, 1)
    plt.imshow(Xk, cmap="gray")
    plt.show()

    compression_values[i] = 1 - (k*(m+n+1)/(m*n))
    i += 1

plt.subplot(1, 2, 2)
plt.plot(k_values, compression_values, 'o')
plt.title('A plot of compression of the image')
plt.xlabel('K Values')
plt.ylabel('Compression')
plt.grid()
